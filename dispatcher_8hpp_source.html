288  template <typename F>
289  std::bitset<CUBOS_CORE_DISPATCHER_MAX_CONDITIONS> Dispatcher::assignConditionBit(F func)
290  {
291  // Check if this condition already exists
292  auto it = std::find_if(mConditions.begin(), mConditions.end(),
293  [&func](const std::shared_ptr<AnySystemWrapper<bool>>& condition) {
294  auto wrapper = dynamic_cast<SystemWrapper<F>*>(condition.get());
295  if (wrapper == nullptr)
296  {
297  return false;
298  }
299  return wrapper->mSystem == func;
300  });
301  if (it != mConditions.end())
302  {
303  // If it does, return the set bit
304  return std::bitset<CUBOS_CORE_DISPATCHER_MAX_CONDITIONS>(1)
305  << static_cast<std::size_t>(it - mConditions.begin());
306  }
307  // If we already reached the condition limit, exit
308  if (mConditions.size() >= CUBOS_CORE_DISPATCHER_MAX_CONDITIONS)
309  {
310  CUBOS_ERROR("The condition limit ({}) was reached!", CUBOS_CORE_DISPATCHER_MAX_CONDITIONS);
311  abort();
312  }
313  // Otherwise, add it to the conditions list
314  mConditions.push_back(std::make_shared<SystemWrapper<F>>(func));
315  // And return the bit it has
316  return std::bitset<CUBOS_CORE_DISPATCHER_MAX_CONDITIONS>(1) << (mConditions.size() - 1);
317  }
