Metrics

Using the metrics system.

See also

Full source code here.

Registering metrics and profiling in CUBOS. can done through the metric macros defined in core/metrics.hpp and the Metrics class. In order to enable these features, compile with -DENABLE_PROFILING.

#include <cubos/core/metrics.hpp>
using cubos::core::Metrics;
    CUBOS_METRIC("a", 1);
    CUBOS_METRIC("b", 2);
    CUBOS_METRIC("a", 3);
    CUBOS_METRIC("c", 3);

    CUBOS_ASSERT(Metrics::size() == 3); // a b c
    CUBOS_ASSERT(Metrics::sizeByName("a") == 2);

The Metrics core class provides functionality to search new metrics and their value.

    std::size_t seenCount{0};
    std::string name{};

    CUBOS_ASSERT(Metrics::readName(name, seenCount));
    CUBOS_ASSERT(seenCount == 1);
    CUBOS_ASSERT(name == "a");

    CUBOS_ASSERT(Metrics::readName(name, seenCount));
    CUBOS_ASSERT(seenCount == 2);
    CUBOS_ASSERT(name == "b");

    CUBOS_ASSERT(Metrics::readName(name, seenCount));
    CUBOS_ASSERT(seenCount == 3);
    CUBOS_ASSERT(name == "c");

    CUBOS_ASSERT(!Metrics::readName(name, seenCount)); // no more new metrics!
    double value;
    std::size_t offset = 0;

    CUBOS_ASSERT(Metrics::readValue("a", value, offset));
    CUBOS_ASSERT(value == 1.0);
    CUBOS_ASSERT(offset == 1);

    CUBOS_ASSERT(Metrics::readValue("a", value, offset));
    CUBOS_ASSERT(value == 3.0);
    CUBOS_ASSERT(offset == 2);

    offset = 0; // use a different offset or reset current one to 0
    CUBOS_ASSERT(Metrics::readValue("b", value, offset));
    CUBOS_ASSERT(value == 2.0);
    CUBOS_ASSERT(offset == 1);

    // etc..

You can calculate how much time a function/scope takes by using CUBOS_PROFILE :

void myFunction()
{
    CUBOS_PROFILE();
    std::this_thread::sleep_for(std::chrono::seconds(2));
    // this function should take about 2000ms...
}
    myFunction();

    offset = 0;
    CUBOS_ASSERT(Metrics::readName(name, seenCount));
    CUBOS_ASSERT(Metrics::readValue(name, value, offset));
    CUBOS_ASSERT(value > 2000.0);
    CUBOS_DEBUG("{} took {} ms", name, value);

Output:

[19:16:43.236] [main.cpp:78 main] debug: "profiling:/home/user/cubos/core/samples/metrics/main.cpp:myFunction:14" took 2000.1806 ms

There's also possibility to limit metric entries to avoid memory usage

    Metrics::clear();
    CUBOS_ASSERT(Metrics::size() == 0);

    Metrics::setMaxEntries(2);
    CUBOS_METRIC("a", 1);
    CUBOS_METRIC("a", 2);
    CUBOS_METRIC("a", 3);
    CUBOS_ASSERT(Metrics::sizeByName("a") == 2);
    CUBOS_ASSERT(Metrics::size() == 1); // only "a"
