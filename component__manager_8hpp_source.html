202  template <typename T> void ComponentManager::registerComponent()
203  {
204  static_assert(std::is_same<T, typename ComponentStorage<T>::Type::Type>(),
205  "A component can't use a storage for a different component type!");
206  const auto type = std::type_index(typeid(T));
207  if (this->typeToIds.find(type) == this->typeToIds.end())
208  {
209  this->typeToIds[type] = this->entries.size() + 1; // Component ids start at 1.
210  this->entries.emplace_back(new typename ComponentStorage<T>::Type());
211  }
