266  template <typename F>
267  typename impl::SystemTraits<F>::Return SystemWrapper<F>::call(World& world, CommandBuffer& commands) const
268  {
269  using Arguments = typename impl::SystemTraits<F>::Arguments;
270  using Fetcher = impl::SystemFetcher<Arguments>;
271 
272  // 1. Fetch the arguments from the world (ReadResource, etc).
273  // 2. Convert the fetched data into the actual arguments (e.g: ReadResource<R> to const R&)
274  // 3. Pass it into the system.
275  auto fetched = Fetcher::fetch(world, commands);
276  auto args = Fetcher::arg(std::move(fetched));
277  return std::apply(this->system, std::forward<Arguments>(args));
278  }
