247  template <typename F> void SystemWrapper<F>::call(const World& world, Commands& commands) const
248  {
249  using Arguments = typename impl::SystemTraits<F>::Arguments;
250  using Fetcher = impl::SystemFetcher<Arguments>;
251 
252  // 1. Fetch the arguments from the world (ReadResource, etc).
253  // 2. Convert the fetched data into the actual arguments (e.g: ReadResource<R> to const R&)
254  // 3. Pass it into the system.
255  auto fetched = Fetcher::fetch(world, commands);
256  auto args = Fetcher::arg(world, std::move(fetched));
257  std::apply(this->system, std::forward<Arguments>(args));
258  }
