223  template <typename F> void SystemWrapper<F>::call(const World& world, Commands& commands) const
224  {
225  using Arguments = typename impl::SystemTraits<F>::Arguments;
226  using Fetcher = impl::SystemFetcher<Arguments>;
227 
228  // 1. Fetch the arguments from the world (ReadResource, etc).
229  // 2. Convert the fetched data into the actual arguments (e.g: ReadResource<R> to const R&)
230  // 3. Pass it into the system.
231  auto fetched = Fetcher::fetch(world, commands);
232  auto args = Fetcher::arg(world, std::move(fetched));
233  std::apply(this->system, std::forward<Arguments>(args));
234  }
