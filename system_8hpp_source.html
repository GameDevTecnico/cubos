321  template <typename F>
322  typename impl::SystemTraits<F>::Return SystemWrapper<F>::call(World& world, CommandBuffer& commands)
323  {
324  using Arguments = typename impl::SystemTraits<F>::Arguments;
325  using Fetcher = impl::SystemFetcher<Arguments>;
326 
327  if (!mState.has_value())
328  {
329  CUBOS_CRITICAL("System was not prepared");
330  abort();
331  }
332 
333  // 1. Fetch the arguments from the world (ReadResource, etc).
334  // 2. Convert the fetched data into the actual arguments (e.g: ReadResource<R> to const R&)
335  // 3. Pass it into the system.
336  auto fetched = Fetcher::fetch(world, commands, mState.value());
337  auto args = Fetcher::arg(std::move(fetched));
338  return std::apply(mSystem, std::forward<Arguments>(args));
339  }
