135  {
136  this->mutex.lock();
137  auto pos = this->stream.tell();
138  this->stream.seek(0, memory::SeekOrigin::Begin);
139  auto des = data::BinaryDeserializer(this->stream);
140  for (auto name : this->names)

142  ComponentType type;
143  des.read(type, &map);
144  commands.add(map.getRef(name), std::move(type));
145  }
